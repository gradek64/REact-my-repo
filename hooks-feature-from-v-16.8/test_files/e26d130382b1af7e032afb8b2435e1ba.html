let nextTodoId = 0;
const todosApiFactory = ({ state, setState }) => {
const addTodo = text => {
setState(prevState => ({
...prevState,
todos: [
...prevState.todos,
{
id: nextTodoId++,
text,
completed: false
}
]
}));
};
const toggleTodo = id => {
setState(prevState => ({
...prevState,
todos: prevState.todos.map(todo => {
if (todo.id === id) {
return {
...todo,
completed: !todo.completed
}
}
return todo
})
}));
};
const setFilter = filter => {
setState(prevState => ({
...prevState,
filter
}));
};
const getFilter = () => {
return state.filter;
};
const getVisibleTodos = () => {
return filterTodos(state.todos, getFilter());
};
return {
addTodo,
toggleTodo,
setFilter,
getFilter,
getVisibleTodos
};
}
function filterTodos(todos, filter) {
switch (filter) {
case "SHOW_ALL":
return todos;
case "SHOW_COMPLETED":
return todos.filter(t => t.completed);
case "SHOW_ACTIVE":
return todos.filter(t => !t.completed);
default:
throw new Error("Unknown filter: " + filter);
}
}

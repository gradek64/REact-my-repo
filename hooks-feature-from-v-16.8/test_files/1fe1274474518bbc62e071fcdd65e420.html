const counterApiFactory = ({ state, setState }) => {
const increment = () => {
setState(prevState => {
if (prevState.frozen) {
return prevState;
}
return {
...prevState,
count: prevState.count + 1
};
});
};
const setFrozen = frozen => {
setState(prevState => ({
...prevState,
frozen
}));
};
const count = state.count;
const frozen = state.frozen;
return {
increment,
setFrozen,
count,
frozen
};
};
const StateCounterV4 = props => {
const counterApi = useApi(counterApiFactory, {
count: 0,
frozen: false
});
useEffect(() => {
counterApi.increment();
counterApi.setFrozen(true);
counterApi.increment();
}, []);
return <div>{counterApi.count}</div>;
};

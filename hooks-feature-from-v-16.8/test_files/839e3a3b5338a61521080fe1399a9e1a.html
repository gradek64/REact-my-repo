let useCounterApi = () => {
let [state, setState] = useState({
count: 0,
frozen: false
});
const increment = () => {
setState(prevState => {
if (prevState.frozen) {
return prevState;
}
return {
...prevState,
count: prevState.count + 1
};
});
};
const setFrozen = frozen => {
setState(prevState => ({
...prevState,
frozen
}));
};
const count = state.count;
const frozen = state.frozen;
return {
increment,
setFrozen,
count,
frozen
};
};
const StateCounterV3 = props => {
let counterApi = useCounterApi();
useEffect(() => {
counterApi.increment();
counterApi.setFrozen(true);
counterApi.increment();
}, []);
return <div>{counterApi.count}</div>;
};

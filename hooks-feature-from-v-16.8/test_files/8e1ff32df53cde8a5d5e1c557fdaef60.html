const StateCounterV1 = () => {
let [count, setCount] = useState(0);
let [frozen, setFrozen] = useState(false);
const increment = () => {
setCount(prevCount => {
if (frozen) {
return prevCount;
}
return prevCount + 1;
});
};
useEffect(() => {
increment();
setFrozen(true);
increment();
}, []);
return <div>{count}</div>;
};
